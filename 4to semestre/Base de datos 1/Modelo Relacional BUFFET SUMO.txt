Table Sucursal{
  Id_Sucursal varchar(20) [not null, pk]
  Nombre varchar(50) [not null, unique]
  Telefono int(10) [not null, unique]
  Descripcion varchar(150) [null]
  Email varchar(50) [not null]
  Imagen varchar(50) [not null]
  Direccion varchar(50) [not null]
  NombreCalle varchar(50) [not null]
  NoCalle varchar(50) [not null]
  CodigoPostal varchar(50) [not null]
  Id_Empleado varchar(18) [not null, ref: < Empleado.Id_Empleado]
  Id_Usuario int [not null, ref: < Usuario.Id_Usuario]
  Id_Cliente int [not null, ref: < Cliente.Id_Cliente]
  Id_Publicidad varchar(50) [not null, ref: < Publicidad.Id_Publicidad]
  Id_AspiranteEmpleado varchar(30) [not null, ref: < AspiranteEmpleo.Id_AspiranteEmpleo]
  Id_Inmobiliario int [not null, ref: < Inmobiliario.Id_Inmobiliario]
  Id_Proveedor varchar(20) [not null, ref: < Proveedor.Id_Proveedor]
  
}


Table Cliente{
  Id_Cliente int [not null, pk]
  Nombre varchar(50) [null]
  Apellido varchar(50) [null]
  ClienteFrecuente boolean [not null]
  Id_Servicios int [not null]
  Id_Menu varchar(50) [not null, ref: - Menu.Id_Menu]
  Id_Cuenta int [not null, ref: < Cuenta.Id_Cuenta]
  NoPedido vachar(50) [not null, ref: < Pedido.NoPedido]
  Id_Reservacion varchar(50) [not null, ref: < Reservacion.Id_Reservacion]
  
}

Table Menu{
  Id_Menu varchar(50) [not null, pk]
  NombreAlimento varchar(150) [not null]
  Cantidad int [not null]
  Descripcion varchar(150) [not null]
  Imagen varchar(150) [not null]
  Precio int [not null]
  Restricciones varchar(150) [null]
  PoliticaConsumo varchar(300) [null]
  indexes{
    (NombreAlimento, Cantidad) [unique, name: 'uk_nombreycantidad']
  }
}

Table Pedido{
  NoPedido int [not null, pk]
  NombreAlimento varchar(150) [not null]
  Especificacion varchar(150) [null]
  Cantidad int [not null]
  Descuento int [null]
  Precio int [not null]
  ProductoNombre varchar(50) [not null, ref: < Producto.Nombre]
  
  
}

Table Categoria{
  Id_Categoria int [not null, unique, pk]
  TipoProducto varchar(50) [not null, ref: > Pedido.Especificacion]
}

Table Producto{ 
  Nombre varchar(50) [not null, pk]
  Categoria varchar(50) [not null]
  Descripcion varchar(150) [not null]
  Imagen varchar(150) [not null]
  Precio int [not null]
  Producto_Nombre varchar(50) [not null, ref: < Ingrediente.Id_Ingrediente]
}

Table Ingrediente{
  Id_Ingrediente varchar(50) [not null, pk]
  CantidadIngrediente int [not null]
}

Table Proveedor{
  Id_Proveedor varchar(20) [not null, pk]
  Id_Producto varchar(20) [not null]
  NombreProveedor varchar(20) [not null]
  ContactoProveedor int [not null]
  DireccionProveedor varchar(20) [not null]
  Id_Ingrediente varchar(50) [not null, ref: < Ingrediente.Id_Ingrediente]
}

Table Cuenta{
  Id_Cuenta varchar(15) [not null, pk]
  NombreAlimento varchar(50) [not null]
  Fecha date [not null]
  Cantidad int [not null]
  PrecioConsumido int [not null]
  Id_MetodoDePago int [not null, ref: < MetodoDePago.Id_MetodoDePago]
}

Table Reservacion{
  Id_Reservacion varchar(50) [not null, pk]
  Correo varchar(50) [not null]
  Nombre varchar(50) [not null]
  Apellido varchar(50) [not null]
  Telefono varchar(50) [not null]
  Dia fecha [not null]
  NumeroPersona int [not null]
  Id_Cliente int [not null]
  Id_Mesa varchar(20) [not null, ref: < Mesa.Id_Mesa]
}

Table Inmobiliario{
  Id_Inmobiliario int [not null, pk]
  TipoDeInmobiliario varchar(50) [not null, ref: < Mesa.Id_Mesa]
}

Table Mesa{
  Id_Mesa varchar (20) [not null, pk]
  Posicion varchar (20) [not null]
  Disponibilidad boolean [not null]
  Capacidad int [not null]
}
